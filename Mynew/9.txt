#include <iostream>
#include <math.h>
using namespace std;

int ploshad(int a){
    return (a*a);
}
int ploshad(int a, int b){
    return ((a+b)*2);
}
float ploshad(float a, float b, float c){
    float p = (a + b + c)/2; return (sqrt (p*(p-a)*(p-b)*(p-c)));
}

int max(int a, int b){
    if (a<b) return b; else return a;
}
int max(int a, int b, int c){
    if (a<b){
        if (c<b) return b;
    }if (c<a) return a;
    else return c;
}
int max(int a, int b, int c, int d){
    if (a<b){
        if (c<b){
            if (d<b) return b;
        }
    }if (c<a){
        if (d<a) return a;
    }if (d<c) return c;
    else return d;
    
}

int min(int a, int b, int c){
    if (a>b){
        if (c>b) return b;
    }if (c>a) return a;
    else return c;
}
int min(int a, int b, int c, int d){

        if (c>b){

        }

float sred(float a, float b){
    return ((a+b)/2);
}
float sred(float a, float b, float c){
    return ((a+b+c)/3);
}
float sred(float a, float b, float c, float d){
    return ((a+b+c+d)/4);
}

{
    //
    cout << "площадь квадрата со стороной 4 = " << ploshad(4) << endl;
